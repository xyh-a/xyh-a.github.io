本周学习了哈夫曼树的性质以及一些作用还有图的一些基础概念
作用
1.哈夫曼树主要运用于数据压缩，哈夫曼编码是实现数据压缩的重要方案
2.哈夫曼编码可以在用二进制给字符编码时使电文总长最短且意思唯一
3贪心算法也跟哈夫曼树有关（不过并未学习贪心算法）
关于哈夫曼树的一些概念和性质
1哈夫曼树其实是最优二叉树，就是带权路径长（wpl）最短的树
2哈夫曼树中权越大的结点离根越近

代码实现

#include<iostream>
using namespace std;
typedef struct huffmantree {
	int weight;//权值
	int lchild;//左孩子
	int rchild;//右孩子
	int parent;//双亲节点
}ht;//建立哈夫曼树结点
ht* huffmanTree = NULL;
void createhuffmantree(int[], int);
void creathuffamncode(int);
int main() {
	int n;
	int w[1000];//接收数据的数组
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> w[i];
	}
	int m = 2 * n - 1;//哈夫曼树的容量
	huffmanTree = new ht[m + 1];//申请m+1大小的空间存储哈夫曼树
	createhuffmantree(w, n);
	creathuffamncode(n);
}
//找到两个最小的权值
void select(ht huffmanTree[], int n, int& s1, int& s2) {
	int min;
	//遍历节点，找出单结点

	for (int i = 1; i <= n; i++) {
		if (huffmanTree[i].parent == 0) {
			min = i;
			break;
		}
	}
	//接着遍历全部节点，找出权值最小的单结点
	for (int i = 1; i <= n; i++) {
		if (huffmanTree[i].parent == 0) {
			if (huffmanTree[i].weight < huffmanTree[min].weight) {
				min = i;
			}
		}
	}
	s1 = min;
	//进行和上面相同的操作，找到第二小的结点
	for (int i = 1; i <= n; i++) {
		if (huffmanTree[i].parent == 0&&i!=s1) {
			min = i;
			break;
		}
	}
	for (int i = 1; i <= n; i++) {
		if (huffmanTree[i].parent == 0 && i != s1) {
			if (huffmanTree[i].weight < huffmanTree[min].weight) {
				min = i;
			}
		}
	}
	s2 = min;
}
//构建哈夫曼树
void createhuffmantree(int w[], int n) {
	int s1;
	int s2;
	int m = 2 * n - 1;
	//1——n号空间存放叶子结点，初始化结点
	for (int i = 0; i <= n; i++) {
		//其中叶子结点的权值是w[n]数组保存
		huffmanTree[i].weight = w[i];
		huffmanTree[i].parent = 0;
		huffmanTree[i].lchild = 0;
		huffmanTree[i].rchild = 0;
	}
	//对于其它的节点即非叶子结点
	for (int i = n + 1; i <= m; i++) {
		huffmanTree[i].weight = w[i];
		huffmanTree[i].parent = 0;
		huffmanTree[i].lchild = 0;
		huffmanTree[i].rchild = 0;
	}
	//创建非叶子结点，构建哈夫曼树
	for (int i = n + 1; i <= m; i++) {
		//找到权值最小的两个结点，分别赋值给s1和s2
		select(huffmanTree, i - 1, s1, s2);
		huffmanTree[s1].parent = i;
		huffmanTree[s2].parent = i;
		huffmanTree[i].lchild = s1;
		huffmanTree[i].rchild = s2;
		huffmanTree[i].weight = huffmanTree[s1].weight + huffmanTree[s2].weight;
	}
}
void creathuffamncode(int n) {
	string* huffmancode = new string[n + 1];
	for (int i = 1; i <= n; i++) {
		string cd = "";
		//我们从叶子结点开始向上遍历
		for (int c = i, p = huffmanTree[i].parent; p != 0; c = p, p = huffmanTree[p].parent) {
			if (huffmanTree[p].lchild == c) {
				cd = "0" + cd;
			}
			else {
				cd = "1" + cd;
			}
		}
		huffmancode[i] = cd;
	}
	//输出编码
	for (int i = 1; i <= n; i++) {
		cout << huffmanTree[i].weight << " " << huffmancode[i] << endl;
	}
}
